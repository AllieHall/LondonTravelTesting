    <h2>Currency Converter</h2>

    <input type="number" id="amount" placeholder="Enter Amount">

    <button onclick="convert('USDtoGBP')">Convert USD to GBP</button>

    <button onclick="convert('GBPtoUSD')">Convert GBP to USD</button>

    <h3 id="result">Result: -</h3>


    <script>

        const rate = 0.79; // Example rate: 1 USD = 0.79 GBP


        function convert(direction) {

            let amount = document.getElementById('amount').value;

            let resultDisplay = document.getElementById('result');

            

            if (direction === 'USDtoGBP') {

                resultDisplay.innerText = "Result: £" + (amount * rate).toFixed(2);

            } else {

                resultDisplay.innerText = "Result: $" + (amount / rate).toFixed(2);

            }

        }

    </script>
